import pandas as pd
import matplotlib.pyplot as plt

# Load data
df = pd.read_csv(r"PremierLeagueMatches.csv")

# Aggregate home vs. away performance
home_stats = df.groupby('Home Team').agg({'homeScore': 'sum', 'homeXG': 'sum'}).reset_index()
away_stats = df.groupby('Away Team').agg({'awayScore': 'sum', 'awayXG': 'sum'}).reset_index()

# Merge and calculate efficiency
merged = home_stats.merge(away_stats, left_on='Home Team', right_on='Away Team')
merged['Home Efficiency'] = merged['homeScore'] / merged['homeXG']
merged['Away Efficiency'] = merged['awayScore'] / merged['awayXG']

# Plot
plt.figure(figsize=(12, 6))
merged.set_index('Home Team')[['Home Efficiency', 'Away Efficiency']].plot(kind='bar', alpha=0.7)
plt.title('Goal Efficiency (Goals/xG) by Venue')
plt.ylabel('Efficiency Ratio')
plt.axhline(1, color='red', linestyle='--', label='Par Efficiency')
plt.legend()
plt.savefig("Goal Efficiency.png")
plt.show()

import pandas as pd
import matplotlib.pyplot as plt

# Load data
df = pd.read_csv(r'PremierLeagueMatches.csv')

# Clean "Attendance" column (remove commas and convert to numeric)
df['Attendance'] = df['Attendance'].str.replace(',', '').astype(float)

# Calculate max attendance per stadium
stadium_max = df.groupby('Stadium')['Attendance'].transform('max')

# Calculate attendance ratio (actual / max capacity)
df['Attendance Ratio'] = df['Attendance'] / stadium_max

# Mark home wins
df['Home Win'] = (df['Result'] == 'H').astype(int)

# Plot
plt.figure(figsize=(10, 5))
df.groupby(pd.cut(df['Attendance Ratio'], bins=[0, 0.7, 0.9, 1]),observed=True)['Home Win'].mean().plot(kind='bar')
plt.title('Home Win Rate by Attendance Ratio')
plt.ylabel('Win Rate')
plt.xlabel('Attendance Ratio')
plt.xticks(rotation=0)
plt.savefig("Home Win Rate by Attendance Ratio.png")
plt.show()
